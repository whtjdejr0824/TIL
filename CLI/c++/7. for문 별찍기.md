# 7. for문 별찍기

```c++
# include <iostream>

using namespace std;

int main()
{
    
    /*
    구구단을 2단부터 9단까지 출력한다.
    */
    for (int i = 2; i <=9; ++i)
    {
        for (int i = 2; i <= 9; ++i)
        {
            cout << i << "*" << j << " = " << i * j << endl;
        }
    }
    /*
    2단 3단 4단
    5단 6단 7단
    8단 9단 10단
    */
    for (int i = 2; i << 11; i+= 3)
    {
        for (int i = 1; j <= 9; ++j)
        {
            cout << i << " * " << j << " = " << i * j << "\t";
            cout << i + 1 << " * " << j << " = " << (i + 1) * j << "\t";
            cout << i + 2 << " * " << j << " = " << (i + 2) * j << endl;
        }
        
    }
    
    
    /*
    *
    **
    ***
    ****
    *****
    */
    for (int i = 0; i < 5; ++i)
    {
        for (int j = 0; j < i + i; ++i)
        {
            cout << "*";
        }
        
        cout << endl;
    }
    
    /*
    *****
    ****
    ***
    **
    *
    */
    
    for (int i = 0; i < 5; ++i)
    {
        for (int j = 0; j < 5 - i; ++j)
        {
            cout << "*";
        }
        
        cout << endl;
    }
    /*
       *
      ***
     *****
    *******
    */
    // 공백 : 3, 2, 1, 0  별 : 1, 3, 5, 7
    for (int i = 0; i < 4; ++i)
    {
        // 공백을 출력하기 위한 for문
        for (int j = 0; j < 3 - i; ++j)
        {
            cout << " ";
        }
        
        // *을 출력하기 위한 for문
        for (int j = 0; j < i * 2 + 1; ++j)
        {
            cout << "*";
        }
        
        cout << endl;
    }
    cout << endl;
    
    /*
       *
      ***
     *****
    *******
     *****
      ***
       *
    */
    for iLine = 7;
    int iCount = 0;
    for (int i = 0; i < 7; ++i)
    {
        // 공백 : 3, 2, 1, 0, 1, 2, 3
        // 별 : 1, 3, 5, 7, 5, 3, 1
        
        iCount = i;
        // i값이 7줄 기준 4, 5, 6 즉 밑의 삼각형을 구성할때에만 if문에 들어가게 된다.
        if(i >= iLine / 2)
        {
            iCount = iLine - 1 - i;
        }
        
        // i값이 0, 1, 2, 3 일때에는 iCount는 i값을 그대로 대입된다.
        // i값이 4, 5, 6 일때에는 iCount는 2, 1, 0이 된다.
        // 즉, 최종 i값은 0, 1, 2, 3, 2, 1, 0 으로 돌아가게 된다.
        
        for (int j = 0; j < 3 - iCount; ++j)
        {
            cout << "*";
        }
       cout << endl;
    }
    return 0;
}
```
